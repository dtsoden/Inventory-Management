# Plan 1: Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold the Next.js project with OOP base classes, encrypted SQLite database, multi-tenant auth, and a setup wizard that configures the platform on first launch.

**Architecture:** Next.js 15 App Router with TypeScript. Prisma ORM over SQLite with tenant-scoped middleware. All secrets stored AES-256-GCM encrypted in a SystemConfig table. Base service/repository classes enforce DRY patterns across all future modules. Setup wizard replaces .env configuration.

**Tech Stack:** Next.js 15, TypeScript, Prisma, SQLite, NextAuth.js v5, ShadCN/ui, Tailwind CSS, Argon2id (via argon2), AES-256-GCM (via Node.js crypto), class-variance-authority (cva)

**Plan Series:** This is Plan 1 of 7. Plans 2-7 build on this foundation.

**Spec Reference:** `docs/superpowers/specs/2026-04-10-inventory-management-platform-design.md`

```prisma
generator client {

datasource db {

// ============================================================
// Platform Configuration
// ============================================================

model SetupState {
  id                    Int       @id @default(1)
  isSetupComplete       Boolean   @default(false)
  setupCompletedAt      DateTime?
  setupCompletedByUserId String?
  encryptionKeySalt     String    // Base64-encoded Argon2id salt
  encryptionKeyHash     String    // Base64-encoded hash for vault key verification
  createdAt             DateTime  @default(now())
  updatedAt             DateTime  @updatedAt
}

model SystemConfig {
  id          String   @id @default(uuid())
  key         String   @unique
  value       String   // AES-256-GCM encrypted for secrets, plaintext otherwise
  isSecret    Boolean  @default(false)
  category    String   // "ai", "email", "integrations", "platform", "cors"
  description String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([category])
}

// ============================================================
// Multi-Tenancy
// ============================================================

model Tenant {
  id        String   @id @default(uuid())
  name      String
  slug      String   @unique
  logoUrl   String?
  settings  String?  // JSON string for branding/preferences
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  users              User[]
  vendors            Vendor[]
  itemCategories     ItemCategory[]
  items              Item[]
  purchaseOrders     PurchaseOrder[]
  assets             Asset[]
  receivingSessions  ReceivingSession[]
  auditLogs          AuditLog[]
  notifications      Notification[]
  chatConversations  ChatConversation[]
}

// ============================================================
// Users & Auth
// ============================================================

model User {
  id           String    @id @default(uuid())
  tenantId     String
  email        String
  passwordHash String
  firstName    String
  lastName     String
  role         String    // SUPER_ADMIN, ORG_ADMIN, MANAGER, WAREHOUSE_STAFF
  avatarUrl    String?
  isActive     Boolean   @default(true)
  lastLoginAt  DateTime?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt

  tenant             Tenant              @relation(fields: [tenantId], references: [id])
  createdOrders      PurchaseOrder[]     @relation("CreatedOrders")
  approvedOrders     PurchaseOrder[]     @relation("ApprovedOrders")
  assignedAssets     Asset[]             @relation("AssignedAssets")
  receivingSessions  ReceivingSession[]
  auditLogs          AuditLog[]
  notifications      Notification[]
  chatConversations  ChatConversation[]

  @@unique([tenantId, email])
  @@index([tenantId])
}

// ============================================================
// Vendors
// ============================================================

model Vendor {
  id          String   @id @default(uuid())
  tenantId    String
  name        String
  contactName String?
  email       String
  phone       String?
  address     String?
  city        String?
  state       String?
  zip         String?
  country     String?
  website     String?
  notes       String?
  isActive    Boolean  @default(true)
  rating      Float?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  tenant         Tenant          @relation(fields: [tenantId], references: [id])
  items          Item[]
  purchaseOrders PurchaseOrder[]

  @@index([tenantId])
}

// ============================================================
// Items & Categories
// ============================================================

model ItemCategory {
  id          String   @id @default(uuid())
  tenantId    String
  name        String
  description String?
  iconName    String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  tenant Tenant @relation(fields: [tenantId], references: [id])
  items  Item[]

  @@index([tenantId])

model Item {
  id          String   @id @default(uuid())
  tenantId    String
  vendorId    String
  categoryId  String?
  name        String
  sku         String?
  description String?
  unitPrice   Float
  imageUrl    String?
  externalId  String?
  isActive    Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  tenant             Tenant              @relation(fields: [tenantId], references: [id])
  vendor             Vendor              @relation(fields: [vendorId], references: [id])
  category           ItemCategory?       @relation(fields: [categoryId], references: [id])
  purchaseOrderLines PurchaseOrderLine[]
  assets             Asset[]

  @@index([tenantId])
  @@index([vendorId])
}

// ============================================================
// Purchase Orders
// ============================================================

model PurchaseOrder {
  id           String    @id @default(uuid())
  tenantId     String
  orderNumber  String
  vendorId     String
  createdById  String
  approvedById String?
  status       String    @default("DRAFT") // DRAFT, PENDING_APPROVAL, APPROVED, SUBMITTED, PARTIALLY_RECEIVED, RECEIVED, CANCELLED
  reason       String?
  subtotal     Float     @default(0)
  tax          Float     @default(0)
  total        Float     @default(0)
  pdfUrl       String?
  submittedAt  DateTime?
  approvedAt   DateTime?
  notes        String?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt

  tenant            Tenant              @relation(fields: [tenantId], references: [id])
  vendor            Vendor              @relation(fields: [vendorId], references: [id])
  createdBy         User                @relation("CreatedOrders", fields: [createdById], references: [id])
  approvedBy        User?               @relation("ApprovedOrders", fields: [approvedById], references: [id])
  lines             PurchaseOrderLine[]
  assets            Asset[]
  receivingSessions ReceivingSession[]

  @@unique([tenantId, orderNumber])
  @@index([tenantId])
  @@index([vendorId])
  @@index([status])
}

model PurchaseOrderLine {
  id               String @id @default(uuid())
  purchaseOrderId  String
  itemId           String
  quantity         Int
  unitPrice        Float
  totalPrice       Float
  quantityReceived Int    @default(0)
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt

  purchaseOrder PurchaseOrder @relation(fields: [purchaseOrderId], references: [id], onDelete: Cascade)
  item          Item          @relation(fields: [itemId], references: [id])

  @@index([purchaseOrderId])
}

// ============================================================
// Assets / Inventory
// ============================================================

model Asset {
  id              String    @id @default(uuid())
  tenantId        String
  itemId          String
  purchaseOrderId String?
  assetTag        String
  serialNumber    String?
  status          String    @default("AVAILABLE") // AVAILABLE, ASSIGNED, IN_MAINTENANCE, RETIRED, LOST
  assignedToId    String?
  assignedAt      DateTime?
  location        String?
  notes           String?
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt

  tenant        Tenant         @relation(fields: [tenantId], references: [id])
  item          Item           @relation(fields: [itemId], references: [id])
  purchaseOrder PurchaseOrder? @relation(fields: [purchaseOrderId], references: [id])
  assignedTo    User?          @relation("AssignedAssets", fields: [assignedToId], references: [id])

  @@unique([tenantId, assetTag])
  @@index([tenantId])
  @@index([status])
}

  tenant        Tenant        @relation(fields: [tenantId], references: [id])
  purchaseOrder PurchaseOrder @relation(fields: [purchaseOrderId], references: [id])
  receivedBy    User          @relation(fields: [receivedById], references: [id])

  @@index([tenantId])

model AuditLog {
  id         String   @id @default(uuid())

  userId     String

  entityType String

  user   User   @relation(fields: [userId], references: [id])

  @@index([tenantId])
  @@index([entityType, entityId])
  @@index([createdAt])

model Notification {
```

```typescript
import { createCipheriv, createDecipheriv, randomBytes, createHash } from 'crypto';

const ALGORITHM = 'aes-256-gcm';
const IV_LENGTH = 12;       // 96 bits for GCM
const TAG_LENGTH = 16;      // 128-bit auth tag
const SALT_LENGTH = 16;     // 128-bit salt
const KEY_LENGTH = 32;      // 256-bit key
const ENCODING = 'base64' as const;

/**
 * Centralized encryption service using AES-256-GCM.
 *
 * All encrypt/decrypt operations in the application flow through this class.
 * No other code should import Node.js crypto directly for data encryption.
 */
export class EncryptionService {
  /**
   * Derive a 256-bit encryption key from a passphrase using PBKDF2.
   *
   * Note: We use PBKDF2 (available in Node.js core) with high iterations
   * rather than Argon2id to avoid native dependency issues in containers.
   * 600,000 iterations of SHA-512 provides strong brute-force resistance.
   */
  async deriveKey(passphrase: string, salt: string): Promise<Buffer> {
    const { pbkdf2 } = await import('crypto');
    return new Promise((resolve, reject) => {
      const saltBuffer = Buffer.from(salt, ENCODING);
      pbkdf2(passphrase, saltBuffer, 600_000, KEY_LENGTH, 'sha512', (err, derivedKey) => {
        if (err) reject(err);
        else resolve(derivedKey);
      });
    });
  }

  /**
   * Encrypt plaintext using AES-256-GCM.
   *
   * Output format: base64(IV + ciphertext + authTag)
   * This is a single opaque string safe for database storage.
   */
  encrypt(plaintext: string, key: Buffer): string {
    const iv = randomBytes(IV_LENGTH);
    const cipher = createCipheriv(ALGORITHM, key, iv, { authTagLength: TAG_LENGTH });

    const encrypted = Buffer.concat([
      cipher.update(plaintext, 'utf8'),
      cipher.final(),
    ]);

    const authTag = cipher.getAuthTag();

    // Pack: IV (12) + ciphertext (variable) + authTag (16)
    const packed = Buffer.concat([iv, encrypted, authTag]);
    return packed.toString(ENCODING);
  }

  /**
   * Decrypt a value previously encrypted with encrypt().
   *
   * Throws if the key is wrong, data is tampered, or format is invalid.
   */
  decrypt(encryptedValue: string, key: Buffer): string {
    const packed = Buffer.from(encryptedValue, ENCODING);

    if (packed.length < IV_LENGTH + TAG_LENGTH) {
      throw new Error('Invalid encrypted data: too short');
    }

    const iv = packed.subarray(0, IV_LENGTH);
    const authTag = packed.subarray(packed.length - TAG_LENGTH);
    const ciphertext = packed.subarray(IV_LENGTH, packed.length - TAG_LENGTH);

    const decipher = createDecipheriv(ALGORITHM, key, iv, { authTagLength: TAG_LENGTH });
    decipher.setAuthTag(authTag);

    const decrypted = Buffer.concat([
      decipher.update(ciphertext),
      decipher.final(),
    ]);

    return decrypted.toString('utf8');
  }

  /**
   * Generate a cryptographically random salt for key derivation.
   * Returns base64-encoded string for database storage.
   */
  static generateSalt(): string {
    return randomBytes(SALT_LENGTH).toString(ENCODING);
  }

  /**
   * Create a deterministic hash of the derived key for vault verification.
   *
   * Used to verify the passphrase on application start without storing the key.
   * SHA-256 of the key itself (not the passphrase).
   */
  hashForVerification(key: Buffer): string {
    return createHash('sha256').update(key).digest(ENCODING);
  }
}
```

```typescript
import { PrismaClient } from '@prisma/client';
import { EncryptionService } from '../encryption/EncryptionService';

interface SetConfigOptions {
  isSecret: boolean;
  category: string;
  description?: string;
}

/**
 * Centralized configuration service.
 *
 * All application configuration reads and writes flow through this class.
 * Secrets are transparently encrypted/decrypted using the vault key.
 * No other code should query SystemConfig directly.
 */
export class ConfigService {
  private vaultKey: Buffer | null = null;

  constructor(
    private readonly prisma: PrismaClient,
    private readonly encryption: EncryptionService,
  ) {}

  /**
   * Set the vault key for encrypting/decrypting secrets.
   * Called once on application start after vault unlock.
   */
  setVaultKey(key: Buffer): void {
    this.vaultKey = key;
  }

  /**
   * Check if the vault is unlocked (key is available in memory).
   */
  isVaultUnlocked(): boolean {
    return this.vaultKey !== null;
  }

  /**
   * Clear the vault key from memory (for logout/shutdown).
   */
  lockVault(): void {
    if (this.vaultKey) {
      this.vaultKey.fill(0); // Zero out the key before releasing
      this.vaultKey = null;
    }
  }

  /**
   * Get a configuration value by key.
   * Automatically decrypts secret values.
   * Returns null if the key does not exist.
   */
  async get(key: string): Promise<string | null> {
    const config = await this.prisma.systemConfig.findUnique({
      where: { key },
    });

    if (!config) return null;

    if (config.isSecret) {
      this.requireVaultKey();
      return this.encryption.decrypt(config.value, this.vaultKey!);
    }

    return config.value;
  }

  /**
   * Set a configuration value.
   * Automatically encrypts secret values before storage.
   */
  async set(key: string, value: string, options: SetConfigOptions): Promise<void> {
    let storedValue = value;

    if (options.isSecret) {
      this.requireVaultKey();
      storedValue = this.encryption.encrypt(value, this.vaultKey!);
    }

    await this.prisma.systemConfig.upsert({
      where: { key },
      create: {
        key,
        value: storedValue,
        isSecret: options.isSecret,
        category: options.category,
        description: options.description,
      },
      update: {
        value: storedValue,
        isSecret: options.isSecret,
        category: options.category,
        description: options.description,
      },
    });
  }

  /**
   * Get all configuration values for a category.
   * Returns a key-value map with secrets decrypted.
   */
  async getByCategory(category: string): Promise<Record<string, string>> {
    const configs = await this.prisma.systemConfig.findMany({
      where: { category },
    });

    const result: Record<string, string> = {};

    for (const config of configs) {
      if (config.isSecret) {
        this.requireVaultKey();
        result[config.key] = this.encryption.decrypt(config.value, this.vaultKey!);
      } else {
        result[config.key] = config.value;
      }
    }

    return result;
  }

  /**
   * Delete a configuration key.
   */
  async delete(key: string): Promise<void> {
    await this.prisma.systemConfig.delete({ where: { key } });
  }

  /**
   * Check if the vault key is available, throw if not.
   */
  private requireVaultKey(): void {
    if (!this.vaultKey) {
      throw new Error('Vault is locked. Provide the encryption passphrase to unlock.');
    }
  }
}
```

```typescript
import { PrismaClient } from '@prisma/client';
import { NotFoundError, TenantIsolationError } from '../errors';
import type { PaginationParams, PaginatedResult, SortParams } from '../types';

interface FindAllOptions {
  pagination?: PaginationParams;
  sort?: SortParams;
  where?: Record<string, unknown>;
  include?: Record<string, boolean | object>;
}

/**
 * Base repository providing tenant-scoped CRUD operations.
 *
 * Every domain repository (VendorRepository, AssetRepository, etc.)
 * extends this class. Tenant isolation is enforced here, not in each
 * individual repository. Write once, inherit many times.
 *
 * Subclasses only need to set `modelName` to the Prisma model accessor.
 */
export abstract class BaseRepository<T extends { id: string; tenantId?: string }> {
  constructor(protected readonly prisma: PrismaClient) {}

  /**
   * The Prisma model name (e.g., 'vendor', 'asset', 'purchaseOrder').
   * Must match the property name on the PrismaClient instance.
   */
  protected abstract get modelName(): string;

  /**
   * Access the Prisma model delegate for this repository.
   */
  protected get model(): any {
    return (this.prisma as any)[this.modelName];
  }

  /**
   * Find all entities for a tenant with optional pagination, sorting, and filtering.
   */
  async findAll(
    tenantId: string,
    options: FindAllOptions = {},
  ): Promise<PaginatedResult<T>> {
    const { pagination, sort, where = {}, include } = options;

    const whereClause = { tenantId, ...where };

    const queryArgs: any = { where: whereClause };

    if (sort) {
      queryArgs.orderBy = { [sort.field]: sort.direction };
    }

    if (include) {
      queryArgs.include = include;
    }

    if (pagination) {
      const { page, pageSize } = pagination;
      queryArgs.skip = (page - 1) * pageSize;
      queryArgs.take = pageSize;

      const [data, total] = await Promise.all([
        this.model.findMany(queryArgs),
        this.model.count({ where: whereClause }),
      ]);

      return {
        data,
        total,
        page,
        pageSize,
        totalPages: Math.ceil(total / pageSize),
      };
    }

    const [data, total] = await Promise.all([
      this.model.findMany(queryArgs),
      this.model.count({ where: whereClause }),
    ]);

    return {
      data,
      total,
      page: 1,
      pageSize: total,
      totalPages: 1,
    };
  }

  /**
   * Find a single entity by ID, scoped to tenant.
   * Returns null if not found or if it belongs to a different tenant.
   */
  async findById(tenantId: string, id: string, include?: Record<string, boolean | object>): Promise<T | null> {
    const queryArgs: any = { where: { id } };
    if (include) {
      queryArgs.include = include;
    }

    const entity = await this.model.findUnique(queryArgs);

    if (!entity) return null;
    if (entity.tenantId && entity.tenantId !== tenantId) return null;

    return entity;
  }

  /**
   * Create a new entity, automatically injecting the tenantId.
   */
  async create(tenantId: string, data: Omit<Partial<T>, 'id' | 'tenantId'>): Promise<T> {
    return this.model.create({
      data: { ...data, tenantId },
    });
  }

  /**
   * Update an entity by ID, verifying tenant ownership first.
   */
  async update(tenantId: string, id: string, data: Partial<T>): Promise<T> {
    await this.verifyTenantOwnership(tenantId, id);

    return this.model.update({
      where: { id },
      data,
    });
  }

  /**
   * Soft-delete an entity by setting isActive to false.
   * Hard deletes are intentionally not provided; audit trail requires data retention.
   */
  async softDelete(tenantId: string, id: string): Promise<T> {
    await this.verifyTenantOwnership(tenantId, id);

    return this.model.update({
      where: { id },
      data: { isActive: false },
    });
  }

  /**
   * Count entities matching a filter, scoped to tenant.
   */
  async count(tenantId: string, where: Record<string, unknown> = {}): Promise<number> {
    return this.model.count({
      where: { tenantId, ...where },
    });
  }

  /**
   * Verify that an entity exists and belongs to the given tenant.
   * Throws NotFoundError if missing, TenantIsolationError if wrong tenant.
   */
  private async verifyTenantOwnership(tenantId: string, id: string): Promise<void> {
    const entity = await this.model.findUnique({ where: { id } });

    if (!entity) {
      throw new NotFoundError(this.modelName, id);
    }

    if (entity.tenantId && entity.tenantId !== tenantId) {
      throw new TenantIsolationError();
    }
  }
}
```

```typescript
import { PrismaClient } from '@prisma/client';
import { BaseRepository } from './BaseRepository';
import { NotFoundError } from '../errors';
import type { TenantContext, PaginationParams, SortParams, PaginatedResult } from '../types';

/**
 * Base service providing audited business operations.
 *
 * Every domain service (VendorService, AssetService, etc.) extends this.
 * Audit logging is handled here, not in each service. Subclasses add
 * domain-specific business logic by overriding or extending methods.
 */
export abstract class BaseService<T extends { id: string; tenantId?: string }> {
  constructor(
    protected readonly repository: BaseRepository<T>,
    protected readonly prisma: PrismaClient,
  ) {}

  /**
   * Human-readable entity name for audit logs (e.g., "Vendor", "PurchaseOrder").
   */
  protected abstract get entityName(): string;

  /**
   * List all entities with pagination and sorting.
   */
  async list(
    ctx: TenantContext,
    options: {
      pagination?: PaginationParams;
      sort?: SortParams;
      where?: Record<string, unknown>;
      include?: Record<string, boolean | object>;
    } = {},
  ): Promise<PaginatedResult<T>> {
    return this.repository.findAll(ctx.tenantId, options);
  }

  /**
   * Get a single entity by ID.
   */
  async getById(ctx: TenantContext, id: string, include?: Record<string, boolean | object>): Promise<T | null> {
    return this.repository.findById(ctx.tenantId, id, include);
  }

  /**
   * Get a single entity by ID, throwing if not found.
   */
  async getByIdOrThrow(ctx: TenantContext, id: string, include?: Record<string, boolean | object>): Promise<T> {
    const entity = await this.repository.findById(ctx.tenantId, id, include);
    if (!entity) {
      throw new NotFoundError(this.entityName, id);
    }
    return entity;
  }

  /**
   * Create a new entity with audit logging.
   */
  async create(ctx: TenantContext, data: Omit<Partial<T>, 'id' | 'tenantId'>): Promise<T> {
    const entity = await this.repository.create(ctx.tenantId, data);

    await this.logAudit(ctx, 'CREATED', entity.id, {
      after: entity,
    });

    return entity;
  }

  /**
   * Update an entity with before/after audit logging.
   */
  async update(ctx: TenantContext, id: string, data: Partial<T>): Promise<T> {
    const before = await this.getByIdOrThrow(ctx, id);
    const after = await this.repository.update(ctx.tenantId, id, data);

    await this.logAudit(ctx, 'UPDATED', id, { before, after });

    return after;
  }

  /**
   * Soft-delete an entity with audit logging.
   */
  async delete(ctx: TenantContext, id: string): Promise<T> {
    const before = await this.getByIdOrThrow(ctx, id);
    const result = await this.repository.softDelete(ctx.tenantId, id);

    await this.logAudit(ctx, 'DEACTIVATED', id, { before });

    return result;
  }

  /**
   * Count entities matching optional filters.
   */
  async count(ctx: TenantContext, where: Record<string, unknown> = {}): Promise<number> {
    return this.repository.count(ctx.tenantId, where);
  }

  /**
   * Write an audit log entry. Protected so subclasses can log custom actions.
   */
  protected async logAudit(
    ctx: TenantContext,
    action: string,
    entityId: string,
    details?: { before?: unknown; after?: unknown },
  ): Promise<void> {
    await this.prisma.auditLog.create({
      data: {
        tenantId: ctx.tenantId,
        userId: ctx.userId,
        action,
        entityType: this.entityName,
        entityId,
        details: details ? JSON.stringify(details) : null,
      },
    });
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getServerSession } from 'next-auth';
import { authOptions } from '../auth-options';
import { AppError, UnauthorizedError, ForbiddenError, isAppError } from '../errors';
import type { SessionUser, TenantContext, UserRole, ApiResponse } from '../types';

type HttpMethod = 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';

interface HandlerOptions {
  requiredRoles?: UserRole[];
}

type RouteHandler = (
  req: NextRequest,
  ctx: TenantContext,
  params?: Record<string, string>,
) => Promise<NextResponse>;

/**
 * Base API route handler providing auth, role checks, tenant context, and error formatting.
 *
 * Usage in route files:
 *
 *   const handler = new VendorApiHandler();
 *   export const GET = handler.handle('GET');
 *   export const POST = handler.handle('POST', { requiredRoles: ['ORG_ADMIN'] });
 *
 * Subclasses implement onGet(), onPost(), etc. for their specific logic.
 * Auth, tenant context, and error handling are provided by this base class.
 */
export abstract class BaseApiHandler {
  /**
   * Create a Next.js route handler with auth, role checks, and error formatting.
   */
  handle(method: HttpMethod, options: HandlerOptions = {}) {
    return async (req: NextRequest, context?: { params: Promise<Record<string, string>> }) => {
      try {
        // 1. Authenticate
        const session = await getServerSession(authOptions);
        if (!session?.user) {
          throw new UnauthorizedError();
        }

        const user = session.user as SessionUser;

        // 2. Authorize (role check)
        if (options.requiredRoles && !options.requiredRoles.includes(user.role)) {
          throw new ForbiddenError(
            `Role ${user.role} does not have permission for this operation`,
          );
        }

        // 3. Build tenant context
        const tenantCtx: TenantContext = {
          tenantId: user.tenantId,
          userId: user.id,
          role: user.role,
        };

        // 4. Resolve route params
        const params = context?.params ? await context.params : undefined;

        // 5. Dispatch to method handler
        const handler = this.getMethodHandler(method);
        return await handler(req, tenantCtx, params);
      } catch (error) {
        return this.formatError(error);
      }
    };
  }

  /**
   * Override these in subclasses to handle specific HTTP methods.
   */
  protected async onGet(req: NextRequest, ctx: TenantContext, params?: Record<string, string>): Promise<NextResponse> {
    return this.methodNotAllowed('GET');
  }

  protected async onPost(req: NextRequest, ctx: TenantContext, params?: Record<string, string>): Promise<NextResponse> {
    return this.methodNotAllowed('POST');
  }

  protected async onPut(req: NextRequest, ctx: TenantContext, params?: Record<string, string>): Promise<NextResponse> {
    return this.methodNotAllowed('PUT');
  }

  protected async onPatch(req: NextRequest, ctx: TenantContext, params?: Record<string, string>): Promise<NextResponse> {
    return this.methodNotAllowed('PATCH');
  }

  protected async onDelete(req: NextRequest, ctx: TenantContext, params?: Record<string, string>): Promise<NextResponse> {
    return this.methodNotAllowed('DELETE');
  }

  /**
   * Helper: create a success JSON response.
   */
  protected success<T>(data: T, status = 200): NextResponse {
    const body: ApiResponse<T> = { success: true, data };
    return NextResponse.json(body, { status });
  }

  /**
   * Helper: create a success response with a message.
   */
  protected successMessage(message: string, status = 200): NextResponse {
    const body: ApiResponse = { success: true, message };
    return NextResponse.json(body, { status });
  }

  /**
   * Helper: parse pagination params from URL search params.
   */
  protected getPagination(req: NextRequest) {
    const searchParams = req.nextUrl.searchParams;
    const page = Math.max(1, parseInt(searchParams.get('page') || '1', 10));
    const pageSize = Math.min(100, Math.max(1, parseInt(searchParams.get('pageSize') || '20', 10)));
    return { page, pageSize };
  }

  /**
   * Helper: parse sort params from URL search params.
   */
  protected getSort(req: NextRequest, allowedFields: string[]) {
    const searchParams = req.nextUrl.searchParams;
    const field = searchParams.get('sortBy');
    const direction = searchParams.get('sortDir') === 'desc' ? 'desc' : 'asc';

    if (field && allowedFields.includes(field)) {
      return { field, direction: direction as 'asc' | 'desc' };
    }
    return undefined;
  }

  private getMethodHandler(method: HttpMethod): RouteHandler {
    const handlers: Record<HttpMethod, RouteHandler> = {
      GET: this.onGet.bind(this),
      POST: this.onPost.bind(this),
      PUT: this.onPut.bind(this),
      PATCH: this.onPatch.bind(this),
      DELETE: this.onDelete.bind(this),
    };
    return handlers[method];
  }

  private formatError(error: unknown): NextResponse {
    if (isAppError(error)) {
      const body: ApiResponse = {
        success: false,
        error: error.message,
      };
      return NextResponse.json(body, { status: error.statusCode });
    }

    console.error('Unhandled error:', error);
    const body: ApiResponse = {
      success: false,
      error: 'Internal server error',
    };
    return NextResponse.json(body, { status: 500 });
  }

  private methodNotAllowed(method: string): NextResponse {
    const body: ApiResponse = {
      success: false,
      error: `Method ${method} not allowed`,
    };
    return NextResponse.json(body, { status: 405 });
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '../db';

/**
 * CORS middleware that reads allowed origins from SystemConfig.
 *
 * All CORS configuration flows from a single source (the 'cors_origins' key
 * in SystemConfig). No per-route CORS configuration exists elsewhere.
 *
 * Values:
 *   "*" = allow all origins (development default)
 *   "https://app.example.com,https://admin.example.com" = specific origins
 */
export async function applyCors(
  req: NextRequest,
  res: NextResponse,
): Promise<NextResponse> {
  const origin = req.headers.get('origin') || '';

  let allowedOrigins = '*';
  try {
    const config = await prisma.systemConfig.findUnique({
      where: { key: 'cors_origins' },
    });
    if (config) {
      allowedOrigins = config.value;
    }
  } catch {
    // During setup (before DB exists), default to open
    allowedOrigins = '*';
  }

  const isAllowed =
    allowedOrigins === '*' ||
    allowedOrigins.split(',').map((o) => o.trim()).includes(origin);

  const effectiveOrigin = allowedOrigins === '*' ? '*' : isAllowed ? origin : '';

  if (effectiveOrigin) {
    res.headers.set('Access-Control-Allow-Origin', effectiveOrigin);
    res.headers.set('Access-Control-Allow-Methods', 'GET, POST, PUT, PATCH, DELETE, OPTIONS');
    res.headers.set('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    res.headers.set('Access-Control-Allow-Credentials', 'true');
    res.headers.set('Access-Control-Max-Age', '86400');
  }

  return res;
}

/**
 * Handle preflight OPTIONS requests.
 */
export function handlePreflight(req: NextRequest): NextResponse | null {
  if (req.method === 'OPTIONS') {
    const res = new NextResponse(null, { status: 204 });
    return res;
  }
  return null;
}
```

```typescript
 * This is the single entry point for all cross-cutting request concerns.
 * CORS, security headers, and setup-redirect logic live here.
 * Do NOT add middleware elsewhere; compose it in this file.
 */
export async function middleware(req: NextRequest) {
  // Handle CORS preflight
  const preflightResponse = handlePreflight(req);
  if (preflightResponse) {
    return applyCors(req, preflightResponse);
  }

  const response = NextResponse.next();

  // Security headers
  response.headers.set('X-Content-Type-Options', 'nosniff');
  response.headers.set('X-Frame-Options', 'DENY');
  response.headers.set('Referrer-Policy', 'strict-origin-when-cross-origin');

  // Apply CORS
  return applyCors(req, response);
}

export const config = {
  matcher: [
    // Match all API routes and pages, skip static files and Next.js internals
    '/((?!_next/static|_next/image|favicon.ico|images/).*)',
  ],
};
```

```typescript
import CredentialsProvider from 'next-auth/providers/credentials';
import { compare } from 'bcryptjs';
import { prisma } from './db';
import type { SessionUser } from './types';

export const authOptions: NextAuthOptions = {
  providers: [
    CredentialsProvider({
      name: 'credentials',
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) {
          return null;
        }

        const user = await prisma.user.findFirst({
          where: {
            email: credentials.email,
            isActive: true,
          },
          include: {
            tenant: { select: { id: true, slug: true, isActive: true } },
          },
        });

        if (!user || !user.tenant.isActive) {
          return null;
        }

        const isPasswordValid = await compare(credentials.password, user.passwordHash);
        if (!isPasswordValid) {
          return null;
        }

        // Update last login
        await prisma.user.update({
          where: { id: user.id },
          data: { lastLoginAt: new Date() },
        });

        return {
          id: user.id,
          email: user.email,
          firstName: user.firstName,
          lastName: user.lastName,
          role: user.role,
          tenantId: user.tenantId,
          tenantSlug: user.tenant.slug,
        } as SessionUser;
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        const sessionUser = user as SessionUser;
        token.id = sessionUser.id;
        token.role = sessionUser.role;
        token.tenantId = sessionUser.tenantId;
        token.tenantSlug = sessionUser.tenantSlug;
        token.firstName = sessionUser.firstName;
        token.lastName = sessionUser.lastName;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        (session.user as SessionUser).id = token.id as string;
        (session.user as SessionUser).role = token.role as any;
        (session.user as SessionUser).tenantId = token.tenantId as string;
        (session.user as SessionUser).tenantSlug = token.tenantSlug as string;
        (session.user as SessionUser).firstName = token.firstName as string;
        (session.user as SessionUser).lastName = token.lastName as string;
      }
      return session;
    },
  },
  pages: {
    signIn: '/login',
  },
  session: {
    strategy: 'jwt',
    maxAge: 8 * 60 * 60, // 8 hours
  },
  secret: process.env.NEXTAUTH_SECRET,
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { hash } from 'bcryptjs';
import { prisma } from '@/lib/db';
import { EncryptionService } from '@/lib/encryption/EncryptionService';
import { ConfigService } from '@/lib/config/ConfigService';
import type { ApiResponse } from '@/lib/types';

const encryptionService = new EncryptionService();

/**
 * GET /api/setup - Check if setup is needed
 */
export async function GET() {
  try {
    const setupState = await prisma.setupState.findUnique({ where: { id: 1 } });
    const isSetupComplete = setupState?.isSetupComplete ?? false;

    return NextResponse.json<ApiResponse<{ isSetupComplete: boolean }>>({
      success: true,
      data: { isSetupComplete },
    });
  } catch {
    // Database may not exist yet on first run
    return NextResponse.json<ApiResponse<{ isSetupComplete: boolean }>>({
      success: true,
      data: { isSetupComplete: false },
    });
  }
}

interface SetupPayload {
  // Step 1: Encryption
  passphrase: string;
  platformName: string;

  // Step 2: Admin
  adminEmail: string;
  adminPassword: string;
  adminFirstName: string;
  adminLastName: string;

  // Step 3: Organization
  orgName: string;
  orgSlug: string;

  // Step 4: Integrations
  openaiApiKey: string;
  smtpHost?: string;
  smtpPort?: string;
  smtpUser?: string;
  smtpPassword?: string;
  catalogApiUrl?: string;

  // Step 5: CORS
  corsOrigins: string; // "*" or comma-separated domains

  // Step 6: Seed data
  seedDemoData: boolean;
}

/**
 * POST /api/setup - Execute platform setup
 */
export async function POST(req: NextRequest) {
  try {
    // Prevent re-setup
    const existingSetup = await prisma.setupState.findUnique({ where: { id: 1 } });
    if (existingSetup?.isSetupComplete) {
      return NextResponse.json<ApiResponse>(
        { success: false, error: 'Setup has already been completed' },
        { status: 409 },
      );
    }

    const payload: SetupPayload = await req.json();

    // Validate required fields
    const required = ['passphrase', 'platformName', 'adminEmail', 'adminPassword',
      'adminFirstName', 'adminLastName', 'orgName', 'orgSlug', 'openaiApiKey', 'corsOrigins'];
    for (const field of required) {
      if (!(payload as any)[field]) {
        return NextResponse.json<ApiResponse>(
          { success: false, error: `Missing required field: ${field}` },
          { status: 400 },
        );
      }
    }

    // 1. Derive encryption key from passphrase
    const salt = EncryptionService.generateSalt();
    const derivedKey = await encryptionService.deriveKey(payload.passphrase, salt);
    const keyHash = encryptionService.hashForVerification(derivedKey);

    // 2. Create setup state
    await prisma.setupState.create({
      data: {
        id: 1,
        isSetupComplete: false,
        encryptionKeySalt: salt,
        encryptionKeyHash: keyHash,
      },
    });

    // 3. Store encrypted config values
    const configService = new ConfigService(prisma, encryptionService);
    configService.setVaultKey(derivedKey);

    await configService.set('platform_name', payload.platformName, {
      isSecret: false, category: 'platform', description: 'Platform display name',
    });

    await configService.set('openai_api_key', payload.openaiApiKey, {
      isSecret: true, category: 'ai', description: 'OpenAI API key',
    });

    await configService.set('cors_origins', payload.corsOrigins, {
      isSecret: false, category: 'cors', description: 'Allowed CORS origins (* or comma-separated domains)',
    });

    // SMTP config (optional)
    if (payload.smtpHost) {
      await configService.set('smtp_host', payload.smtpHost, {
        isSecret: false, category: 'email', description: 'SMTP server hostname',
      });
      await configService.set('smtp_port', payload.smtpPort || '587', {
        isSecret: false, category: 'email', description: 'SMTP server port',
      });
      if (payload.smtpUser) {
        await configService.set('smtp_user', payload.smtpUser, {
          isSecret: false, category: 'email', description: 'SMTP username',
        });
      }
      if (payload.smtpPassword) {
        await configService.set('smtp_password', payload.smtpPassword, {
          isSecret: true, category: 'email', description: 'SMTP password',
        });
      }
    }

    if (payload.catalogApiUrl) {
      await configService.set('catalog_api_url', payload.catalogApiUrl, {
        isSecret: false, category: 'integrations', description: 'External product catalog API URL',
      });
    }

    // 4. Create tenant
    const tenant = await prisma.tenant.create({
      data: {
        name: payload.orgName,
        slug: payload.orgSlug,
        isActive: true,
      },
    });

    // 5. Create admin user
    const passwordHash = await hash(payload.adminPassword, 12);
    await prisma.user.create({
      data: {
        tenantId: tenant.id,
        email: payload.adminEmail,
        passwordHash,
        firstName: payload.adminFirstName,
        lastName: payload.adminLastName,
        role: 'SUPER_ADMIN',
        isActive: true,
      },
    });

    // 6. Generate NEXTAUTH_SECRET if not set
    if (!process.env.NEXTAUTH_SECRET) {
      const { randomBytes } = await import('crypto');
      const secret = randomBytes(32).toString('hex');
      await configService.set('nextauth_secret', secret, {
        isSecret: true, category: 'platform', description: 'NextAuth.js session secret',
      });
    }

    // 7. Mark setup as complete
    await prisma.setupState.update({
      where: { id: 1 },
      data: {
        isSetupComplete: true,
        setupCompletedAt: new Date(),
      },
    });

    return NextResponse.json<ApiResponse>({
      success: true,
      message: 'Setup completed successfully',
    });
  } catch (error) {
    console.error('Setup error:', error);
    return NextResponse.json<ApiResponse>(
      { success: false, error: 'Setup failed. Check server logs for details.' },
      { status: 500 },
    );
  }
}
```

```typescript
  adminFirstName: string;
  adminLastName: string;
  orgName: string;
  orgSlug: string;

  smtpHost: string;
  smtpPort: string;
  smtpUser: string;
  smtpPassword: string;
  catalogApiUrl: string;
  corsOrigins: string;
  seedDemoData: boolean;
}

const INITIAL_DATA: SetupData = {
  passphrase: '',
  platformName: 'Inventory Management Platform',
  adminEmail: '',
  adminPassword: '',
  adminFirstName: '',
  adminLastName: '',
  orgName: '',
  orgSlug: '',
  openaiApiKey: '',
  smtpHost: '',
  smtpPort: '587',
  smtpUser: '',
  smtpPassword: '',
  catalogApiUrl: '',
  corsOrigins: '*',
  seedDemoData: true,
};

const STEP_TITLES = [
  'Welcome',
  'Admin Account',
  'Organization',
  'Integrations',
  'CORS & Security',
  'Review & Launch',
];

export function SetupWizard() {
  const [currentStep, setCurrentStep] = useState(0);
  const [data, setData] = useState<SetupData>(INITIAL_DATA);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const totalSteps = STEP_TITLES.length;
  const progress = ((currentStep + 1) / totalSteps) * 100;

  function updateData(partial: Partial<SetupData>) {
    setData((prev) => ({ ...prev, ...partial }));
  }

  function nextStep() {
    setCurrentStep((prev) => Math.min(prev + 1, totalSteps - 1));
  }

  function prevStep() {
    setCurrentStep((prev) => Math.max(prev - 1, 0));
  }

  async function handleSubmit() {
    setIsSubmitting(true);
    setError(null);

    try {
      const res = await fetch('/api/setup', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data),
      });

      const result = await res.json();

      if (!result.success) {
        setError(result.error || 'Setup failed');
        return;
      }

      window.location.href = '/login';
    } catch (err) {
      setError('Network error. Please try again.');
    } finally {
      setIsSubmitting(false);
    }
  }

  const stepProps = { data, updateData, onNext: nextStep, onBack: prevStep };

  return (
    <Card className="w-full max-w-2xl shadow-card-active">
      <CardHeader className="space-y-4">
        <div className="flex items-center justify-between">
          <CardTitle className="text-xl font-semibold">
            {STEP_TITLES[currentStep]}
          </CardTitle>
          <span className="text-sm text-muted-foreground">
            Step {currentStep + 1} of {totalSteps}
          </span>
        </div>
        <Progress value={progress} className="h-2" />
      </CardHeader>
      <CardContent>
        {error && (
          <div className="mb-4 rounded-md bg-destructive/10 p-3 text-sm text-destructive">
            {error}
          </div>
        )}

        {currentStep === 0 && <WelcomeStep {...stepProps} />}
        {currentStep === 1 && <AdminStep {...stepProps} />}
        {currentStep === 2 && <OrgStep {...stepProps} />}
        {currentStep === 3 && <IntegrationsStep {...stepProps} />}
        {currentStep === 4 && <CorsStep {...stepProps} />}
        {currentStep === 5 && (
          <ReviewStep
            {...stepProps}
            onSubmit={handleSubmit}
            isSubmitting={isSubmitting}
          />
        )}
      </CardContent>
    </Card>
  );
}
```

```typescript
import { signIn } from 'next-auth/react';
import { useRouter } from 'next/navigation';

import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Loader2 } from 'lucide-react';

export default function LoginPage() {
  const router = useRouter();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
```
